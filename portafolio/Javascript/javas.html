class  Persona {
    constructor ( nombre , apellido ) {
        esto . _nombre = nombre ;
        esto . _a Apellido = apellido ;
    }
    obtener  nombre ( ) {
        devuelve  esto . _nombre ;
    }
    set  nombre ( nombre ) {
        esto . _nombre = nombre ;
    }
    obtener  apellido ( ) {
        devuelve  esto . _a Apellidos ;
    }
    establecer  apellido ( apellido ) {
        esto . _a Apellido = apellido ;
    }
    nombreCompleto ( ) {
        devuelve  esto . _nombre + '' + esto . _a Apellidos ; 
    }
}
